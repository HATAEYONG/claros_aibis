## 2. 백엔드 배포

### 2.1 의존성 설치

```bash
cd backend

# 가상 환경 생성
python -m venv venv
source venv/bin/activate  # Linux/macOS
# venv\Scripts\activate   # Windows

# 의존성 설치
pip install -r requirements/production.txt
```

---

### 2.2 환경 변수 설정

```bash
# .env 파일 생성
cat > .env << EOF
# Django
SECRET_KEY=your-secret-key-here
DEBUG=False
ALLOWED_HOSTS=claros-aibis.com,www.claros-aibis.com

# Database
DB_ENGINE=django.db.backends.postgresql
DB_NAME=claros_aibis
DB_USER=claros_user
DB_PASSWORD=your-db-password
DB_HOST=your-db-host.claros-aibis.com
DB_PORT=5432

# CORS
CORS_ALLOWED_ORIGINS=https://claros-aibis.com,https://www.claros-aibis.com

# Security
JWT_SECRET_KEY=your-jwt-secret-key
JWT_ACCESS_TOKEN_LIFETIME=60  # minutes
JWT_REFRESH_TOKEN_LIFETIME=1440  # minutes (24 hours)
EOF
```

---

### 2.3 데이터베이스 마이그레이션

```bash
# 마이그레이션 생성
python manage.py makemigrations

# 마이그레이션 적용
python manage.py migrate

# 데이터 시딩
python manage.py seed_data

# 슈퍼유저 생성
python manage.py createsuperuser
```

---

### 2.4 정적 파일 수집

```bash
python manage.py collectstatic --noinput
```

---

### 2.5 Gunicorn 설정

```bash
# gunicorn.conf.py
cat > gunicorn.conf.py << EOF
bind = "0.0.0.0:8000"
workers = 4
worker_class = "sync"
worker_connections = 1000
max_requests = 1000
max_requests_jitter = 100
timeout = 30
keepalive = 5
preload_app = True
accesslog = "/var/log/claros_aibis/access.log"
errorlog = "/var/log/claros_aibis/error.log"
loglevel = "info"
EOF

# Gunicorn 실행
gunicorn config.wsgi:application -c gunicorn.conf.py
```

---

### 2.6 Supervisor 설정

```bash
# /etc/supervisor/conf.d/claros_aibis.conf
cat > /etc/supervisor/conf.d/claros_aibis.conf << EOF
[program:claros_aibis]
directory=/var/www/claros_aibis/backend
command=/var/www/claros_aibis/venv/bin/gunicorn config.wsgi:application -c gunicorn.conf.py
user=www-data
autostart=true
autorestart=true
redirect_stderr=true
stdout_logfile=/var/log/claros_aibis/supervisor.log
EOF

# Supervisor 시작
sudo supervisorctl reread
sudo supervisorctl update
sudo supervisorctl start claros_aibis
```

### 3.4 Nginx 설정

```bash
# /etc/nginx/sites-available/claros_aibis
cat > /etc/nginx/sites-available/claros_aibis << EOF
# Backend API
server {
    listen 80;
    server_name api.claros-aibis.com;

    location / {
        proxy_pass http://127.0.0.1:8000;
        proxy_set_header Host \$host;
        proxy_set_header X-Real-IP \$remote_addr;
        proxy_set_header X-Forwarded-For \$proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto \$scheme;
    }

    location /static/ {
        alias /var/www/claros_aibis/backend/static/;
    }

    location /media/ {
        alias /var/www/claros_aibis/backend/media/;
    }
}

# Frontend
server {
    listen 80;
    server_name claros-aibis.com www.claros-aibis.com;

    root /var/www/claros_aibis/frontend/dist;
    index index.html;

    location / {
        try_files \$uri \$uri/ /index.html;
    }

    location ~* \.(js|css|png|jpg|jpeg|gif|ico|svg|woff|woff2|ttf|eot)$ {
        expires 1y;
        add_header Cache-Control "public, immutable";
    }
}
EOF

# 심볼릭 링크 생성
sudo ln -s /etc/nginx/sites-available/claros_aibis /etc/nginx/sites-enabled/

# Nginx 테스트 및 재시작
sudo nginx -t
sudo systemctl restart nginx
```

## 4. Docker 배포 (선택사항)

### 4.1 docker-compose.yml

```yaml
version: '3.8'

services:
  postgres:
    image: postgres:14
    environment:
      POSTGRES_DB: claros_aibis
      POSTGRES_USER: claros_user
      POSTGRES_PASSWORD: ${DB_PASSWORD}
    volumes:
      - postgres_data:/var/lib/postgresql/data
    ports:
      - "5432:5432"

  backend:
    build: ./backend
    command: gunicorn config.wsgi:application -c gunicorn.conf.py
    volumes:
      - ./backend:/app
      - static_volume:/app/static
      - media_volume:/app/media
    environment:
      - DEBUG=False
      - DB_HOST=postgres
    depends_on:
      - postgres

  frontend:
    build: ./frontend
    volumes:
      - ./frontend/dist:/usr/share/nginx/html
    ports:
      - "3000:80"

  nginx:
    image: nginx:alpine
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - ./nginx.conf:/etc/nginx/nginx.conf
      - static_volume:/static
      - media_volume:/media
      - ./certbot/conf:/etc/letsencrypt
      - ./certbot/www:/var/www/certbot
    depends_on:
      - backend
      - frontend

volumes:
  postgres_data:
  static_volume:
  media_volume:
```

---

### 4.2 Docker 실행

```bash
# 빌드
docker-compose build

# 실행
docker-compose up -d

# 로그 확인
docker-compose logs -f

# 중지
docker-compose down
```

## 5. CI/CD 파이프라인

### 5.1 GitHub Actions

```yaml
# .github/workflows/deploy.yml
name: Deploy

on:
  push:
    branches: [main]

jobs:
  backend-test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - name: Set up Python
        uses: actions/setup-python@v4
        with:
          python-version: '3.11'
      - name: Install dependencies
        run: |
          pip install -r requirements/production.txt
      - name: Run tests
        run: pytest

  frontend-test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - name: Set up Node
        uses: actions/setup-node@v3
        with:
          node-version: '18'
      - name: Install dependencies
        run: npm install
      - name: Run tests
        run: npm test

  deploy:
    needs: [backend-test, frontend-test]
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - name: Deploy to server
        uses: appleboy/ssh-action@master
        with:
          host: ${{ secrets.HOST }}
          username: ${{ secrets.USERNAME }}
          key: ${{ secrets.SSH_KEY }}
          script: |
            cd /var/www/claros_aibis
            git pull origin main
            ./deploy.sh
```

## 6. 배포 절차

### 6.1 일반 배포

```bash
# 1. 코드 풀
git pull origin main
```
