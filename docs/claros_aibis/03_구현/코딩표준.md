**ViewSet 구조**:
```python
class MasterAccountViewSet(ExportImportMixin, BatchOperationsMixin, viewsets.ModelViewSet):
    """
    마스터 계정과목 ViewSet
    """
    serializer_class = MasterAccountSerializer
    permission_classes = [IsAuthenticated]
    lookup_field = 'account_id'

    def get_queryset(self):
        """쿼리셋 반환"""
        queryset = MasterAccount.objects.all()
        account_type = self.request.query_params.get('account_type')
        if account_type:
            queryset = queryset.filter(account_type=account_type)
        return queryset

    @action(detail=False, methods=['get'])
    def account_types(self, request):
        """계정 유형 목록 조회"""
        types = [choice[0] for choice in MasterAccount.ACCOUNT_TYPE_CHOICES]
        return Response({'account_types': types})
```

```typescript
function useApi<T>(
  endpoint: string,
  params?: Record<string, any>
) {
  const [data, setData] = useState<T | null>(null);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<Error | null>(null);

  const fetchData = async () => {
    setLoading(true);
    setError(null);
    try {
      const response = await api.get(endpoint, { params });
      setData(response.data);
    } catch (err) {
      setError(err as Error);
    } finally {
      setLoading(false);
    }
  };

  useEffect(() => {
    fetchData();
  }, [endpoint, JSON.stringify(params)]);

  return { data, loading, error, refetch: fetchData };
}
```

```typescript
// src/services/masterDataService.ts
import api from './api';

export const accountAPI = {
  getAccounts: async (params?: Record<string, any>) => {
    const response = await api.get('/data-hub/master/accounts/', { params });
    return response.data;
  },

  getAccount: async (id: number) => {
    const response = await api.get(`/data-hub/master/accounts/${id}/`);
    return response.data;
  },

  createAccount: async (data: Partial<MasterAccount>) => {
    const response = await api.post('/data-hub/master/accounts/', data);
    return response.data;
  },

  updateAccount: async (id: number, data: Partial<MasterAccount>) => {
    const response = await api.put(`/data-hub/master/accounts/${id}/`, data);
    return response.data;
  },

  deleteAccount: async (id: number) => {
    await api.delete(`/data-hub/master/accounts/${id}/`);
  },

  exportAccounts: async (format: 'csv' | 'excel' | 'json') => {
    const response = await api.post('/data-hub/master/accounts/export/', { format });
    return response.data;
  },
};
```

### 3.2 KISS (Keep It Simple, Stupid)

```typescript
// ❌ 나쁜 예 (복잡)
const isValid = (data: any) => {
  if (data && data.account_id && typeof data.account_id === 'number' && data.account_id > 0) {
    if (data.account_code && typeof data.account_code === 'string' && data.account_code.length > 0) {
      return true;
    }
  }
  return false;
};

// ✅ 좋은 예 (단순)
const isValid = (data: any): data is MasterAccount => {
  return !!(
    data?.account_id > 0 &&
    typeof data?.account_code === 'string' &&
    data?.account_code.length > 0
  );
};
```

### 5.1 Python pytest

```python
import pytest
from django.urls import reverse
from rest_framework.test import APIClient

@pytest.mark.django_db
class TestMasterAccountAPI:
    """마스터 계정과목 API 테스트"""

    def test_list_accounts(self, auth_client):
        """계정과목 목록 조회 테스트"""
        response = auth_client.get(reverse('account-list'))
        assert response.status_code == 200
        assert 'results' in response.data

    def test_create_account(self, auth_client):
        """계정과목 생성 테스트"""
        data = {
            'account_code': '110100',
            'account_name': '현금',
            'account_type': 'asset',
        }
        response = auth_client.post(reverse('account-list'), data)
        assert response.status_code == 201
        assert response.data['account_code'] == '110100'

    @pytest.mark.parametrize('account_type', ['asset', 'liability', 'equity'])
    def test_filter_by_type(self, auth_client, account_type):
        """계정 유형별 필터 테스트"""
        response = auth_client.get(
            reverse('account-list'),
            {'account_type': account_type}
        )
        assert response.status_code == 200
```

## 6. Git 커밋 메시지 규칙

### 6.1 커밋 메시지 형식

```
<type>(<scope>): <subject>

<body>

<footer>
```

**Type**:
- feat: 새로운 기능
- fix: 버그 수정
- docs: 문서 변경
- style: 코드 스타일 (기능 변경 없음)
- refactor: 리팩토링
- test: 테스트 추가/수정
- chore: 빌드/설정 변경

**예시**:
